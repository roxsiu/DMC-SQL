-- 0) Creación de base de datos y esquemas

CREATE DATABASE HorasExtrasDB;
USE HorasExtrasDB;
CREATE SCHEMA STG;
CREATE SCHEMA ODS;

-- 1) Tablas dimensionales
CREATE TABLE dbo.DIM_SEDE (
    COD_SEDE CHAR(3) PRIMARY KEY,
    SEDE NVARCHAR(100) NOT NULL
);

CREATE TABLE dbo.DIM_RECLUTAMIENTO (
    COD_RECLU CHAR(3) PRIMARY KEY,
    NOMBRE NVARCHAR(100) NOT NULL
);

CREATE TABLE dbo.DIM_PERSONAL (
    CODIGO CHAR(5) PRIMARY KEY,
    APELLIDO NVARCHAR(100),
    NOMBRE NVARCHAR(100),
    FEC_INICIO DATE,
    COD_SEDE CHAR(3) FOREIGN KEY REFERENCES dbo.DIM_SEDE(COD_SEDE),
    CARGO NVARCHAR(100),
    GENERO NVARCHAR(20),
    COD_RECLU CHAR(3) FOREIGN KEY REFERENCES dbo.DIM_RECLUTAMIENTO(COD_RECLU),
    CodSp NVARCHAR(20) NULL,
    NombreSp NVARCHAR(100) NULL
);

-- 2) Tablas de hechos

CREATE TABLE dbo.FAC_HORAS (
    ID_HORA INT IDENTITY(1,1) PRIMARY KEY,
    EMPLEADO CHAR(5),
    MES_PAGADO DATE,
    MES_TRABAJADO DATE,
    COD_SEDE CHAR(3),
    CARGO NVARCHAR(100),
    SISTEMA_PENSION NVARCHAR(50),
    UN_HE_25 DECIMAL(10,2),
    UN_HE_35 DECIMAL(10,2),
    UN_HE_100 DECIMAL(10,2),
    HE_25 DECIMAL(10,2),
    HE_35 DECIMAL(10,2),
    HE_100 DECIMAL(10,2)
);

CREATE TABLE dbo.FAC_METAS (
    ID_META INT IDENTITY(1,1) PRIMARY KEY,
    ANO INT,
    MES INT,
    MONTO_META DECIMAL(10,2),
    FECHA DATE
);

-- 3) Carga y transformación desde STG
--Las tablas STG se cargan por consola)

INSERT INTO dbo.DIM_SEDE
SELECT LTRIM(RTRIM([CodSede])), LTRIM(RTRIM([Sede])) FROM STG.DIM_SEDE;

INSERT INTO dbo.DIM_RECLUTAMIENTO
SELECT LTRIM(RTRIM([Código])), LTRIM(RTRIM([Nombre])) FROM STG.DIM_RECLUTAMIENTO;

INSERT INTO dbo.DIM_PERSONAL
SELECT 
    LTRIM(RTRIM([Código])),
    LTRIM(RTRIM([Apellido])),
    LTRIM(RTRIM([Nombre])),
    TRY_CONVERT(DATE, [FecInicio], 103),
    LTRIM(RTRIM([CodSede])),
    LTRIM(RTRIM([Cargo])),
    LTRIM(RTRIM([Género])),
    LTRIM(RTRIM([Relutamiento])),
    CodSp,
    NombreSp
FROM STG.DIM_PERSONAL;

INSERT INTO dbo.FAC_METAS
SELECT 
    TRY_CAST([Año] AS INT),
    TRY_CAST([Mes] AS INT),
    TRY_CAST([MontoMeta] AS DECIMAL(10,2)),
    DATEFROMPARTS(TRY_CAST([Año] AS INT), TRY_CAST([Mes] AS INT), 1)
FROM STG.FAC_METAS;

INSERT INTO dbo.FAC_HORAS
SELECT
    LTRIM(RTRIM([Empleado])),
    DATEFROMPARTS(2000 + TRY_CAST(RIGHT(LTRIM(RTRIM([Mes Pagado])),2) AS INT),
        CASE UPPER(LEFT(LTRIM(RTRIM([Mes Pagado])),3))
            WHEN 'ENE' THEN 1 WHEN 'FEB' THEN 2 WHEN 'MAR' THEN 3
            WHEN 'ABR' THEN 4 WHEN 'MAY' THEN 5 WHEN 'JUN' THEN 6
            WHEN 'JUL' THEN 7 WHEN 'AGO' THEN 8 WHEN 'SEP' THEN 9
            WHEN 'OCT' THEN 10 WHEN 'NOV' THEN 11 WHEN 'DIC' THEN 12 END, 1),
    DATEFROMPARTS(2000 + TRY_CAST(RIGHT(LTRIM(RTRIM([Mes Trabajado])),2) AS INT),
        CASE UPPER(LEFT(LTRIM(RTRIM([Mes Trabajado])),3))
            WHEN 'ENE' THEN 1 WHEN 'FEB' THEN 2 WHEN 'MAR' THEN 3
            WHEN 'ABR' THEN 4 WHEN 'MAY' THEN 5 WHEN 'JUN' THEN 6
            WHEN 'JUL' THEN 7 WHEN 'AGO' THEN 8 WHEN 'SEP' THEN 9
            WHEN 'OCT' THEN 10 WHEN 'NOV' THEN 11 WHEN 'DIC' THEN 12 END, 1),
    LTRIM(RTRIM([Sede])),
    LTRIM(RTRIM([Cargo])),
    LTRIM(RTRIM([Sistema Pensión])),
    TRY_CAST([UN HE 25%] AS DECIMAL(10,2)),
    TRY_CAST([UN HE 35%] AS DECIMAL(10,2)),
    TRY_CAST([UN HE 100%] AS DECIMAL(10,2)),
    TRY_CAST([HE 25%] AS DECIMAL(10,2)),
    TRY_CAST([HE 35%] AS DECIMAL(10,2)),
    TRY_CAST([HE 100%] AS DECIMAL(10,2))
FROM STG.FAC_HORAS;


-- 4) Vista integrada de empleados
CREATE OR ALTER VIEW dbo.vw_DIM_EMPLEADO AS
SELECT
    p.CODIGO AS CodEmpleado,
    CONCAT(p.NOMBRE, ' ', p.APELLIDO) AS NombreCompleto,
    p.CARGO,
    p.GENERO,
    p.FEC_INICIO AS FechaInicio,
    s.SEDE,
    r.NOMBRE AS Reclutamiento
FROM dbo.DIM_PERSONAL p
LEFT JOIN dbo.DIM_SEDE s ON p.COD_SEDE = s.COD_SEDE
LEFT JOIN dbo.DIM_RECLUTAMIENTO r ON p.COD_RECLU = r.COD_RECLU;
GO

-- 5) Consultas analíticas

-- 1. Listar empleados con su sede y fecha de inicio

SELECT 
    P.CODIGO,
    P.NOMBRE,
    P.APELLIDO,
    S.SEDE,
    P.FEC_INICIO
FROM DIM_PERSONAL AS P
INNER JOIN DIM_SEDE AS S ON P.COD_SEDE = S.COD_SEDE;

2. Mostrar todas las fechas registradas ordenadas por año y mes

SELECT 
     MES_TRABAJADO
FROM FAC_HORAS
GROUP BY MES_TRABAJADO;

3. Ver todas las horas extras registradas (25%, 50%, 100%)

SELECT 
    MES_TRABAJADO,
    SUM(HE_25 + HE_35 + HE_100) AS TOTAL_HORAS_EXTRA
FROM FAC_HORAS
GROUP BY MES_TRABAJADO;

-- 4. Total de horas extras por empleado

SELECT 
    P.CODIGO,
    P.NOMBRE,
    P.APELLIDO,
    SUM(H.HE_25 + H.HE_35 + H.HE_100) AS MONTO_TOTAL_HEXTRA
FROM FAC_HORAS AS H
INNER JOIN DIM_PERSONAL AS P ON H.EMPLEADO = P.CODIGO
GROUP BY P.CODIGO, P.NOMBRE, P.APELLIDO
ORDER BY 4 DESC;

-- 5. Total de horas extras por sede

SELECT 
    S.SEDE,
    SUM(H.HE_25 + H.HE_35 + H.HE_100) AS TOTAL_HORAS_EXTRA
FROM FAC_HORAS AS H
INNER JOIN DIM_SEDE AS S ON H.SEDE = S.COD_SEDE
GROUP BY S.SEDE;

-- 7. Comparar horas trabajadas vs metas por mes

SELECT 
    M.ANO,
    M.MES,
    M.MONTO_META,
    SUM(H.UN_HE_25 + H.UN_HE_35 + H.UN_HE_100) AS TOTAL_UNIDADES_TRABAJADAS,
    CAST(
        (SUM(H.UN_HE_25 + H.UN_HE_35 + H.UN_HE_100) / NULLIF(M.MONTO_META, 0)) * 100 
        AS DECIMAL(10,2)
    ) AS PORC_CUMPLIMIENTO
FROM dbo.FAC_METAS AS M
LEFT JOIN dbo.FAC_HORAS AS H 
    ON YEAR(H.MES_TRABAJADO) = M.ANO 
    AND MONTH(H.MES_TRABAJADO) = M.MES
GROUP BY M.ANO, M.MES, M.MONTO_META
ORDER BY M.ANO, M.MES;



-- 8. Los 5 empleados con más horas extras

SELECT TOP 5
    P.CODIGO,
    P.NOMBRE,
    P.APELLIDO,
    SUM(H.HE_25 + H.HE_35 + H.HE_100) AS TOTAL_HORAS_EXTRA
FROM FAC_HORAS AS H
INNER JOIN DIM_PERSONAL AS P ON H.EMPLEADO = P.CODIGO
GROUP BY P.CODIGO, P.NOMBRE, P.APELLIDO
ORDER BY TOTAL_HORAS_EXTRA DESC;

 --9. Promedio de horas extras por sede y género

SELECT 
    S.SEDE,
    P.GENERO,
    AVG(H.HE_25 + H.HE_35 + H.HE_100) AS PROMEDIO_HORAS
FROM FAC_HORAS AS H
INNER JOIN DIM_PERSONAL AS P ON H.EMPLEADO = P.CODIGO
INNER JOIN DIM_SEDE AS S ON H.SEDE = S.COD_SEDE
GROUP BY S.SEDE, P.GENERO;

-- 10. Porcentaje de cumplimiento de metas por mes

SELECT 
    M.ANO,
    M.MES,
    M.MONTO_META,
    SUM(H.UN_HE_25 + H.UN_HE_35 + H.UN_HE_100) AS TOTAL_UNIDADES,
    (SUM(H.UN_HE_25 + H.UN_HE_35 + H.UN_HE_100) * 100.0 / M.MONTO_META) AS PORCENTAJE_CUMPLIMIENTO
FROM dbo.FAC_METAS AS M
LEFT JOIN dbo.FAC_HORAS AS H 
    ON YEAR(H.MES_TRABAJADO) = M.ANO 
    AND MONTH(H.MES_TRABAJADO) = M.MES
GROUP BY M.ANO, M.MES, M.MONTO_META
ORDER BY M.ANO, M.MES;
